    static public int[] FibSeriesIterative(int n)
{
    /*
    * e.g.
    * FibSeriesIterative(9);
    * 0, 1, 1, 2, 3, 5, 8, 13, 21...
    */

    int a = 0, b = 1, c = 0;
    int[] data = new int[n];
    int dataPos = 0;

    data[dataPos] = a;
    dataPos++;

    if (n == 0) return data;

    data[dataPos] = b;
    dataPos++;

    if (n == 1) return data;

    for(int i= 2; i < n; i++)
    {
            c = a + b;

            data[dataPos] = c;
            dataPos++;

            a = b;
            b = c;
    }

    return data;
}
