        public static void ShuffleArray(int[] arr)
    {
        int length = arr.Length;
        for(int i = 0; i < length; i++)
        {
            int temp = arr[i];
            int rand = Random.Range(0, length);

            arr[i] = arr[rand];
            arr[rand] = temp;
        }
    }
